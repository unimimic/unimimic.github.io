int readpin = 8;
int status = 0;

void setup() {
  Serial.begin(9600);
  pinMode(readpin, INPUT);
}

void loop() {
  if(digitalRead(readpin)==LOW){
    if (status==1){
      Serial.println("low"); // 回應訊息給電腦
    }
    status = 0;
  }else{
    if (status==0){
      Serial.println("high"); // 回應訊息給電腦
    }
    status = 1;
  }
  delay(10);
}
